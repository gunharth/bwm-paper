// Init firebase auth before Vue inits the App
  firebase.auth.onAuthStateChanged(firebaseUser => {
  if (firebaseUser) {
    store.dispatch("autoSignIn", firebaseUser);
  }
  if (!app) {
    /* eslint-disable no-new */
    app = new Vue({
      router,
      store,
      render: h => h(App)
    }).$mount("#app");
  }
});
